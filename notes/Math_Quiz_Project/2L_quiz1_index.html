ប្រលងតេស្ដការយល់ដឹង៖ មេរៀនបម្លែងលីនេអ៊ែរ (LT)

សូមបំពេញព័ត៌មាន៖

ឈ្មោះ
ល.រ ៖
ក្រុម/ថ្នាក់ ៖

១. យោងតាមនិយមន័យប្រមាណវិធីលើ $\mathcal{L}(V,W)$ តើផលបូកនៃពីរអនុវត្តន៍លីនេអ៊ែរ $(S+T)(v)$ ត្រូវបានកំណត់ដូចម្ដេច?

២. សម្រាប់ $T \in \mathcal{L}(V,W)$ និងស្កាលែ $\lambda \in \mathbb{F}$ តើប្រមាណវិធីគុណស្កាលែ $(\lambda T)(v)$ កំណត់ដោយរូបមន្តមួយណា?

៣. ក្នុងនិយមន័យនៃប្រមាណវិធីបូក $(S+T)(v)$ និងគុណស្កាលែ $(\lambda T)(v)$ តើវ៉ិចទ័រ $v$ ត្រូវយកចេញពីសំណុំមួយណា?

៤. ប្រសិនបើ $S, T \in \mathcal{L}(V,W)$ នោះអនុវត្តន៍ថ្មី $S+T$ ដែលទើបនឹងបង្កើត គឺជាធាតុនៃសំណុំមួយណា?

៥. តើគោលបំណងនៃការកំណត់ប្រមាណវិធីបូក $(S+T)$ និងគុណស្កាលែ $(\lambda T)$ លើសំណុំ $\mathcal{L}(V,W)$ គឺដើម្បីបញ្ជាក់ថា $\mathcal{L}(V,W)$ ជារចនាសម្ព័ន្ធអ្វី?

៦. អនុលោមតាមអ័ក្សយូម (Axioms) តើបម្លែងលីនេអ៊ែរ $T: V \to W$ ត្រូវបំពេញលក្ខខណ្ឌពីរអ្វីខ្លះ?

៧. តើលក្ខខណ្ឌសមមូលតែមួយណា ដែលអាចបញ្ជាក់ថា $T$ ជាអនុវត្តន៍លីនេអ៊ែរបាន?

៨. សម្រាប់បម្លែងលីនេអ៊ែរ $T \in \mathcal{L}(V, W)$ ប្រសិនបើ $V = W$ តើគេហៅធាតុរបស់ $\mathcal{L}(V,V)$ ថាជាអ្វី?

៩. បើ $T$ ជាអនុវត្តន៍លីនេអ៊ែរ តើ $T(0)$ មានតម្លៃស្មើនឹងអ្វី?

១០. តើទំនាក់ទំនងមួយណាដែលត្រឹមត្រូវជានិច្ចសម្រាប់អនុវត្តន៍លីនេអ៊ែរ $T$ ចំពោះគ្រប់ $v \in V$?

១១. នៅក្នុងលំហ $\mathcal{L}(V, W)$ តើផលបូកនៃអនុវត្តន៍លីនេអ៊ែរពីរ $(S+T)(v)$ កំណត់ដោយរូបមន្តមួយណា?

១២. ផ្អែកតាមទ្រឹស្ដីបទ តើសំណុំ $\mathcal{L}(V, W)$ រួមជាមួយប្រមាណវិធីបូកនិងគុណនឹងស្កាលែ បង្កើតបានជារចនាសម្ព័ន្ធអ្វី?

១៣. នៅក្នុងលំហ $\mathcal{L}(V, W)$ តើអ្វីជាធាតុណឺត (Additive identity)?

១៤. តើផលគុណបណ្ដាក់ (Composition) នៃអនុវត្តន៍លីនេអ៊ែរ $(ST)(u)$ កំណត់ដោយរូបមន្តមួយណា?

១៥. ក្នុងចំណោមលក្ខណៈនៃផលគុណបណ្ដាក់របស់អនុវត្តន៍លីនេអ៊ែរ តើលក្ខណៈមួយណាដែល មិនពិតជាទូទៅ (Not true in general)?

១៦. តាង $\varphi_1: \mathbb{R}^2 \longrightarrow \mathbb{R}^2$ ជាអនុវត្តន៍កំណត់ដោយ $\varphi_1((x, y))=(x+y, x-y)$។ តើ $\varphi_1$ ជាអនុវត្តន៍លីនេអ៊ែរដែរឬទេ?

១៧. តាង $\varphi_2: \mathbb{R}^2 \longrightarrow \mathbb{R}^2$ កំណត់ដោយ $\varphi_2((x, y))=(\sin (x+y), \sin (x-y))$។ តើ $\varphi_2$ ជាអនុវត្តន៍លីនេអ៊ែរដែរឬទេ?

១៨. យក $f: V \longrightarrow W$ ជាអនុវត្តន៍លីនេអ៊ែរ។ បើ $S$ ជាសំណុំ Spanning set របស់ $V$ នោះ តើ $f(S)$ ជាអ្វី?

១៩. បើ $f: V \longrightarrow W$ ជាអនុវត្តន៍លីនេអ៊ែរ ហើយ $f(S)$ ជាសំណុំឯករាជ្យលីនេអ៊ែរ (li) ក្នុង $W$ តើគេអាចទាញសេចក្ដីសន្និដ្ឋានអ្វីបានពី $S$?

២០. បើ $S$ ជាសំណុំឯករាជ្យលីនេអ៊ែរ (li) ក្នុង $V$ ហើយ $f$ ជាម៉ូណូម័រភីស (Monomorphism) នោះ តើ $f(S)$ ជាអ្វី?

២១. គេមានបម្លែង $\mathbf{L}: \mathbb{R}^2 \longrightarrow \mathbb{R}^2$ ដែល $\mathbf{L}\left(\begin{bmatrix}1 \\ 1\end{bmatrix}\right)=\begin{bmatrix}2 \\ -3\end{bmatrix}$ និង $\mathbf{L}\left(\begin{bmatrix}0 \\ 1\end{bmatrix}\right)=\begin{bmatrix}1 \\ 2\end{bmatrix}$។ តើ $\mathbf{L}\left(\begin{bmatrix}3 \\ -2\end{bmatrix}\right)$ ស្មើនឹងអ្វី?

២២. កំណត់ $T: \mathbb{R}^3 \rightarrow \mathbb{R}^2$ ដោយ $T(x, y, z)=(2 x-4 y+3 z+b, 6 x+c x y z)$។ តើលក្ខខណ្ឌអ្វីដែលធ្វើឱ្យ $T$ ជាអនុវត្តន៍លីនេអ៊ែរ?

២៣. យក $b, c \in \mathbb{R}$ និង $T: \mathcal{P}(\mathbb{R}) \rightarrow \mathbb{R}^2$ កំណត់ដោយ $T p=\left(3 p(4)+5 p^{\prime}(6)+b p(1) p(2), \int_{-1}^2 x^3 p(x) d x+c \sin p(0)\right)$។ ពេលណាទើប $T$ ជាអនុវត្តន៍លីនេអ៊ែរ?

២៤. គេឲ្យអនុវត្តន៍ $T \in \mathcal{L}(\mathcal{P}(\mathbb{R}), \mathbb{R})$ កំណត់ដោយ $T(p) = \int_0^1 p(x) d x$។ តើការអះអាងមួយណាត្រឹមត្រូវ?

២៥. តាង $T: \mathcal{P}(\mathbb{R}) \rightarrow \mathcal{P}(\mathbb{R})$ កំណត់ដោយ $(T p)(x)=x^2 p(x)$ សម្រាប់គ្រប់ $x \in \mathbb{R}$។ តើ $T$ ជាអ្វី?

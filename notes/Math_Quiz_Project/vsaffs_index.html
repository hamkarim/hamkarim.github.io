ការណែនាំ៖ លំហាត់នេះមានកម្រិតងាយស្រួល សម្រាប់សាកល្បងការយល់ដឹងជាមូលដ្ឋានទៅលើ វ៉ិចទ័រ ការបូកវ៉ិចទ័រ និងសញ្ញាណនៃលំហរង និងលំហអាហ្វីន។
១. នៅក្នុងប្លង់ទ្វេវិមាត្រ (លំហ \(\mathbb{R}^2\)) តើវ៉ិចទ័រមួយត្រូវបានតំណាងដោយអ្វី?
២. ប្រមាណវិធីបូកវ៉ិចទ័រ៖ ប្រសិនបើគេមានវ៉ិចទ័រ $\vec{u} = (1, 2)$ និង $\vec{v} = (3, 1)$ តើផលបូក $\vec{u} + \vec{v}$ ស្មើនឹងប៉ុន្មាន?
៣. ប្រមាណវិធីគុណវ៉ិចទ័រនឹងស្កាលែ (ចំនួនពិត)៖ ប្រសិនបើ $c = 2$ និង $\vec{v} = (-1, 4)$ តើកន្សោម $c\vec{v}$ មានតម្លៃស្មើនឹងអ្វី?
៤. លក្ខណៈនៃវ៉ិចទ័រសូន្យ៖ សម្រាប់គ្រប់វ៉ិចទ័រ $\vec{v}$ នៅក្នុងលំហវ៉ិចទ័រ តើផលបូក $\vec{v} + \vec{0}$ ស្មើនឹងអ្វី?
៥. នៅក្នុងនិយមន័យនៃ "លំហរងវ៉ិចទ័រ" (Vector Subspace) តើលក្ខខណ្ឌចាំបាច់ទី១ ដែលសំណុំមួយត្រូវតែមានជានិច្ច គឺជាអ្វី?
៦. ផ្អែកលើលក្ខខណ្ឌនៃលំហរងខាងលើ តើមួយណាខាងក្រោមតំណាងឱ្យ "លំហរង" (Subspace) នៅក្នុង $\mathbb{R}^2$?
៧. តើវ៉ិចទ័រផ្ទុយ (Additive Inverse) នៃវ៉ិចទ័រ $\vec{u} = (5, -3)$ គឺអ្វី?
៨. ការបូកវ៉ិចទ័រមានលក្ខណៈត្រឡប់ (Commutativity)។ តើសមភាពមួយណាខាងក្រោមដែលបង្ហាញពីលក្ខណៈនេះបានត្រឹមត្រូវ?
៩. តើទម្រង់ទូទៅនៃបន្សំលីនេអ៊ែរ (ប្រើប្រមាណវិធីបូក និង គុណ) រវាងពីរវ៉ិចទ័រ $\vec{u}$ និង $\vec{v}$ ដោយមាន $a$ និង $b$ ជាស្កាលែរ គឺជាអ្វី?
១០. ប្រសិនបើគេយកវ៉ិចទ័រ $\vec{w}$ បូកជាមួយនឹងវ៉ិចទ័រផ្ទុយរបស់វា $(-\vec{w})$ តើលទ្ធផលដែលទទួលបានគឺជាអ្វី?
១១. តាមលក្ខខណ្ឌនៃលំហរងវ៉ិចទ័រ តើសមីការប្លង់ \( P: ax + by + cz = d \) មួយណាទើបអាចចាត់ទុកជា "លំហរង" នៃ \( \mathbb{R}^3 \) បាន?
១២. ក្នុងចំណោមសំណុំរងនៃ \( \mathbb{R}^3 \) ខាងក្រោម តើសំណុំមួយណាដែលជាលំហរង (Vector Subspace) ពិតប្រាកដ?
១៣. ឧបមាថា \( P_1 \) និង \( P_2 \) ជាប្លង់ពីរផ្សេងគ្នាដែលសុទ្ធតែឆ្លងកាត់គល់តម្រុយ \( (0,0,0) \) ក្នុង \( \mathbb{R}^3 \)។ តើប្រពន្ធនៃប្លង់ទាំងពីរនេះ \( P_1 \cap P_2 \) តំណាងឱ្យអ្វី ហើយវាជាលំហរងដែរឬទេ?
១៤. ឧបមាថា \( L_1 \) និង \( L_2 \) ជាបន្ទាត់ពីរផ្សេងគ្នាដែលឆ្លងកាត់គល់តម្រុយក្នុង \( \mathbb{R}^3 \)។ តើប្រជុំនៃបន្ទាត់ទាំងពីរនេះ \( L_1 \cup L_2 \) ជាលំហរងនៃ \( \mathbb{R}^3 \) ដែរឬទេ?
១៥. បន្ទាត់មួយកាត់គល់តម្រុយក្នុង \( \mathbb{R}^3 \) អាចសរសេរជាទម្រង់សំណុំបានថា \[ S = \{ t(a, b, c) \mid t \in \mathbb{R} \} \] តើទម្រង់នេះមានន័យដូចម្តេចក្នុងភាសាពិជគណិតលីនេអ៊ែរ?
១៦. គេឱ្យសំណុំ \[ S = \{ (x, y, z) \in \mathbb{R}^3 \mid x + y + z = 0 \text{ and } x - y = 0 \} \] តើ \( S \) តំណាងឱ្យរូបរាងធរណីមាត្រអ្វី?
១៧. តើលំហរង (Subspace) នៅក្នុងលំហ \( \mathbb{R}^3 \) អាចមានវិមាត្រ (Dimension) ប៉ុន្មានខ្លះ?
១៨. ឧបមាថាប្លង់ \( P \) មួយកាត់តាមគល់តម្រុយក្នុង \( \mathbb{R}^3 \) ដែលមានវ៉ិចទ័រណរម៉ាល់ (Normal Vector) \( \vec{n} = (1, -2, 1) \)។ ក្នុងចំណោមវ៉ិចទ័រខាងក្រោម តើវ៉ិចទ័រមួយណាដែលស្ថិតនៅក្នុងប្លង់លំហរង \( P \) នេះ?
១៩. តើអ្វីជាលក្ខណៈខុសគ្នាសំខាន់បំផុតរវាង "លំហអាហ្វីន" (Affine Space) និង "លំហវ៉ិចទ័រ" (Vector Space)?
២០. នៅក្នុងលំហអាហ្វីន ប្រសិនបើគេយកចំណុចមួយ (Point) ដកនឹងចំណុចមួយទៀត (ឧទាហរណ៍ \( A - B \)) តើលទ្ធផលទទួលបានគឺជាអ្វី?
២១. ឧបមាថា \( A \) និង \( B \) ជាពីរចំណុចក្នុងលំហអាហ្វីន។ តើកន្សោមបន្សំ \( \alpha A + \beta B \) តំណាងឱ្យ "ចំណុចថ្មីមួយទៀត" ក្នុងលំហអាហ្វីនបាន លុះត្រាតែលក្ខខណ្ឌអ្វីត្រូវបានផ្ទៀងផ្ទាត់?
២២. លំហរងអាហ្វីន (Affine Subspace) មួយត្រូវបានកំណត់ដោយធាតុសំខាន់ពីរ។ តើធាតុទាំងពីរនោះគឺជាអ្វី?
២៣. តើសមីការប៉ារ៉ាម៉ែត្រនៃបន្ទាត់អាហ្វីន ដែលកាត់ចំណុច \( A \) និងមានវ៉ិចទ័រប្រាប់ទិស \( \vec{u} \) គឺមានទម្រង់យ៉ាងដូចម្តេច? (\( t \in \mathbb{R} \))
២៤. ឧបមាថាចំណុច ៣ ផ្សេងគ្នា \( A, B, C \) មិនរត់ត្រង់ជួរ (Non-collinear) ក្នុងលំហអាហ្វីន។ តើលំហរងអាហ្វីនដែលបង្កើតដោយចំណុចទាំងបីនេះ (Affine Span) មានរូបរាងធរណីមាត្រជាអ្វី?
២៥. នៅក្នុងលំហអាហ្វីន \( \mathbb{R}^3 \) បន្ទាត់ពីរអាចមានទំនាក់ទំនងនឹងគ្នាជាច្រើនយ៉ាង។ ក្រៅពី កាត់គ្នា ស្របគ្នា និងត្រួតស៊ីគ្នា តើបន្ទាត់ពីរក្នុងលំហ 3D អាចមានស្ថានភាពអ្វីទៀត?
២៦. ចំណុចកណ្តាល \( I \) នៃអង្កត់ \( [AB] \) ក្នុងលំហអាហ្វីន អាចសរសេរបានជាទម្រង់ត្រឹមត្រូវតាមរយៈកន្សោមណា?
២៧. ឧបមាថា \( P_1 \) និង \( P_2 \) ជាប្លង់អាហ្វីនពីរ។ តើលក្ខខណ្ឌអ្វីដែលធ្វើឱ្យប្លង់ទាំងពីរនេះស្របគ្នាដាច់ខាត (Strictly parallel)?
២៨. ប្រសិនបើ \( A, B, C \) ជាចំណុចក្នុងលំហអាហ្វីន ទំនាក់ទំនងហ្សាលនៃវ៉ិចទ័រ ត្រូវបានសរសេរយ៉ាងដូចម្តេច?
២៩. លំហរងអាហ្វីនពីរដែលកាត់គ្នា នឹងបង្កើតបានជាលំហរងអាហ្វីនមួយទៀត។ ប្រសិនបើប្លង់ពីរផ្សេងគ្នាក្នុង \( \mathbb{R}^3 \) កាត់គ្នា តើលំហរងប្រសព្វនោះមានវិមាត្រប៉ុន្មាន?
៣០. នៅក្នុងប្លង់អាហ្វីន \( \mathbb{R}^2 \) តើអ្វីជាភាពខុសគ្នាសំខាន់បំផុតរវាង "ចំណុច" (Point) និង "វ៉ិចទ័រ" (Vector)?
៣១. ឧបមាថា \( A \) និង \( B \) ជាពីរចំណុចផ្សេងគ្នាក្នុងប្លង់អាហ្វីន \( \mathbb{R}^2 \)។ តើកន្សោមបន្សំ \( \alpha A + \beta B \) បង្កើតបានជា "ចំណុចមួយ" (Point) លុះត្រាតែលក្ខខណ្ឌអ្វីត្រូវបានផ្ទៀងផ្ទាត់?
៣២. តើលំហរងអាហ្វីន (Affine Subspace) នៅក្នុង \( \mathbb{R}^2 \) អាចមានវិមាត្រ (Dimension) អ្វីខ្លះ?
៣៣. សមីការបន្ទាត់ទូទៅ \( ax + by = c \) តំណាងឱ្យលំហរងអាហ្វីនមួយនៅក្នុង \( \mathbb{R}^2 \)។ តើក្នុងករណីណាទើបបន្ទាត់នេះ អាចចាត់ទុកជា "លំហរងវ៉ិចទ័រ" (Vector Subspace) ផងដែរ?
៣៤. បន្ទាត់អាហ្វីនមួយឆ្លងកាត់ចំណុច \( A(1, 2) \) និងមានវ៉ិចទ័រប្រាប់ទិស \( \vec{u}(3, -1) \)។ តើសមីការប៉ារ៉ាម៉ែត្ររបស់បន្ទាត់នេះគឺអ្វី? (\( t \in \mathbb{R} \))
៣៥. បន្ទាត់ពីរក្នុងប្លង់អាហ្វីនមានសមីការ \( 2x - y = 3 \) និង \( 4x - 2y = 5 \)។ តើបន្ទាត់ទាំងពីរនេះមានទំនាក់ទំនងយ៉ាងដូចម្តេច?
៣៦. នៅក្នុងប្លង់អាហ្វីន \( \mathbb{R}^2 \) ប្រសិនបើបន្ទាត់ពីរមិនស្របគ្នា (Non-parallel) តើចំណុចប្រសព្វរបស់ពួកវាមានស្ថានភាពបែបណា?
៣៧. ប្រមាណវិធីបំព្រិល (Translation)៖ ប្រសិនបើគេធ្វើបំព្រិលបន្ទាត់អាហ្វីន \( D \) តាមរយៈវ៉ិចទ័រ \( \vec{v} \) (ដែល \( \vec{v} \) មិនមែនជាវ៉ិចទ័រប្រាប់ទិសរបស់ \( D \)) តើលទ្ធផល \( D' \) ទទួលបានគឺជាអ្វី?
៣៨. ដើម្បីបង្កើតបានជា "តម្រុយអាហ្វីន" (Affine Frame / Repère Affine) មួយពេញលេញនៅក្នុងប្លង់ \( \mathbb{R}^2 \) តើគេត្រូវការចំណុចអប្បបរមាចំនួនប៉ុន្មាន?
៣៩. បន្ទាត់អាហ្វីន \( D \) ឆ្លងកាត់ចំណុច \( A \) និង \( B \) (\( A \neq B \))។ តើ "ទិសដៅ" (Direction) នៃបន្ទាត់ \( D \) ដែលតំណាងឱ្យលំហរងវ៉ិចទ័រ ត្រូវបានបង្កើតឡើងដោយអ្វី?
៤០. នៅក្នុងប្លង់អាហ្វីន តើសំណុំនៃចំណុច \( M = (1-t)A + tB \) ដែលលក្ខខណ្ឌ \( 0 \le t \le 1 \) តំណាងឱ្យរូបរាងធរណីមាត្រអ្វី?
៤១. គេឱ្យចំណុច ៤ ក្នុងលំហអាហ្វីន \( A, B, C, D \)។ តើលក្ខខណ្ឌចាំបាច់ និងគ្រប់គ្រាន់ដើម្បីឱ្យចំណុចទាំង ៤ នេះ "មិនអាស្រ័យលីនេអ៊ែរគ្នា" (Affinely Independent) គឺជាអ្វី?
៤២. ក្នុងលំហអាហ្វីន \( \mathbb{R}^3 \), គេមានបន្ទាត់ \( L \) និងប្លង់ \( P \)។ ប្រសិនបើវ៉ិចទ័រប្រាប់ទិស \( \vec{u} \) នៃបន្ទាត់ \( L \) មិនស្ថិតក្នុងលំហរងវ៉ិចទ័រប្រាប់ទិសនៃប្លង់ \( P \) ទេ តើទំនាក់ទំនងរវាង \( L \) និង \( P \) គឺជាអ្វី?
៤៣. ចំណុច \( M \) មួយមានកូអរដោនេបារីសង់ \( (1/2, 1/3, 1/6) \) ធៀបនឹងតម្រុយអាហ្វីន \( (A, B, C) \)។ តើសំណើមួយណាដែលត្រឹមត្រូវ?
៤៤. ប្លង់អាហ្វីនពីរក្នុង \( \mathbb{R}^4 \) ដែលមានវិមាត្រ ២ ដូចគ្នា (ជាប្លង់)។ តើវិមាត្រនៃប្រសព្វរបស់ពួកវា (ប្រសិនបើមិនទទេ) អាចមានតម្លៃអ្វីខ្លះ?
៤៥. ឧបមាថា \( L_1 \) និង \( L_2 \) ជាបន្ទាត់អាហ្វីនពីរក្នុង \( \mathbb{R}^3 \) ដែល "ខ្វែងគ្នា" (Skew lines)។ តើលំហរងអាហ្វីនដែលតូចបំផុតដែលផ្ទុកបន្ទាត់ទាំងពីរនេះ (Affine Hull) មានវិមាត្រប៉ុន្មាន?

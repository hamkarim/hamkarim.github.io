វិញ្ញាសាគណិតវិទ្យា ត្រៀមប្រឡងបាក់ឌុប

ប្រធានបទ៣៖ ការសិក្សាអនុគមន៍ និង ស៊្វីត (កម្រិតខ្ពស់)

កម្រិត៖ មធ្យមសិក្សាទុតិយភូមិ | រយៈពេល៖ ១៥ នាទី

តាង $n$ ជាចំនួនគត់ធម្មជាតិ។ គេមានអនុគមន៍ $f_n$ កំណត់លើ $\mathbb{R}^*$ ដោយ៖ $$f_n(x) = \frac{e^{nx}}{x^2}$$

តាងខ្សែកោង $(\mathscr{C}_n)$ តំណាងអនុគមន៍ $f_n$ ក្នុងតម្រុយអរតូណរម៉ាល់ $(O, \vec{i}, \vec{j})$។

១. គណនា $\displaystyle\lim_{x \to -\infty} f_n(x)$, $\displaystyle\lim_{x \to +\infty} f_n(x)$ និង $\lim_{x \to 0} f_n(x)$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ត្រង់ $-\infty$ ៖ ពិនិត្យមើលសញ្ញានៃ $e^{-\infty}$។
- ត្រង់ $+\infty$ ៖ ប្រៀបធៀបរវាងអនុគមន៍អ៊ិចស្ប៉ូណង់ស្យែល និងពហុធា (អ៊ិចស្ប៉ូណង់ស្យែលកើនលឿនជាង)។
២. សិក្សាមែកអនន្តនៃខ្សែកោង $(\mathscr{C}_n)$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ទាញរកអាស៊ីមតូតឈរ និងអាស៊ីមតូតដេក តាមរយៈចម្លើយលីមីតក្នុងសំណួរទី១។
- ខាង $+\infty$ ត្រូវគណនា $\displaystyle\lim_{x \to +\infty} \frac{f_n(x)}{x}$ បន្ថែម ដើម្បីរកមែកប៉ារ៉ាបូល។
៣. បង្ហាញថា $f_n$ មានដេរីវេលើ $]0; +\infty[$ និង $]-\infty; 0[$ រួចគណនា $f_n'(x)$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ប្រើរូបមន្តដេរីវេផលចែក៖ $\left(\dfrac{u}{v}\right)' = \dfrac{u'v - uv'}{v^2}$ ដោយ $u = e^{nx}$ នាំឱ្យ $u' = ne^{nx}$។
៤. សង់តារាងអថេរភាពនៃ $f_n$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ដាក់កត្តា $e^{nx}$ ចេញ រួចសិក្សាសញ្ញានៃកន្សោមដែលនៅសល់។ កុំភ្លេចថា $x^2 \gt 0$ តែ $x^3$ យកសញ្ញាតាម $x$។
៥. បង្ហាញថាសមីការ $f_n(x) = 1$ មានឫសតែមួយគត់ $x_n$ ក្នុងចន្លោះ $]-\infty; 0[$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ត្រូវប្រើ ទ្រឹស្ដីបទតម្លៃកណ្ដាល៖ បង្ហាញថាអនុគមន៍ជាប់ និង ម៉ូណូតូនដាច់ខាត (កើនដាច់ខាត ឬ ចុះដាច់ខាត) លើចន្លោះនោះ រួចរកទីតាំងនៃលេខ $1$។
៦. បង្ហាញថា $(\forall n \ge 2) \quad -1 \lt x_n \lt -\frac{1}{n}$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - គណនា $f_n(-1)$ និង $f_n(-\frac{1}{n})$ រួចប្រៀបធៀបតម្លៃទាំងពីរនេះជាមួយ $1$ (ដោយចាំថា $f_n(x_n) = 1$)។
៧. បង្ហាញថា $(\forall n \in \mathbb{N}^*) \quad f_{n+1}(x_n) = e^{x_n}$ រួចទាញបញ្ជាក់ថាស៊្វីត $(x_n)_n$ ជាស៊្វីរួម។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ជំនួស $x_n$ ចូលក្នុង $f_{n+1}$ ហើយប្រើសម្មតិកម្ម $e^{n x_n} = x_n^2$ ដែលទាញចេញពី $f_n(x_n) = 1$ ដើម្បីសម្រួលកន្សោម។
- ដើម្បីបង្ហាញថាជាស៊ីតរួម ត្រូវបង្ហាញថាវាជាស៊ីតកើន និងទាល់លើ។ ប្រៀបធៀប $f_{n+1}(x_n)$ ជាមួយ $1$។
៧. ស្រាយបញ្ជាក់ថា $(\forall n \ge 2) \quad x_n \ge -\frac{2\ln n}{n}$ រួចទាញរក $\displaystyle\lim_{n \to +\infty} x_n$ និង $\displaystyle\lim_{n \to +\infty} nx_n$។
💡 គន្លឹះសម្រាប់សិស្ស៖ - ចាប់ផ្តើមពីទំនាក់ទំនង $x_n^2 = e^{n x_n}$ ដោយបំពាក់ $\ln$ សងខាង រួចទាញប្រើវិសមភាពពីសំណួរទី៦ មកជំនួស។
- ប្រើទ្រឹស្ដីបទតម្លៃកៀប (Théorème des gendarmes) ដើម្បីទាញរកលីមីត។
៨. សង់ខ្សែកោង $(\mathscr{C}_1)$។

កំណែលំហាត់៣៖ ការសិក្សាអនុគមន៍ និង ស៊្វីត

សូមប្អូនៗពិនិត្យចម្លើយពេញលេញជាជំហានៗ

បម្រាប់លំហាត់

យក $n \in \mathbb{N}^*$ និងកំណត់អនុគមន៍ $f_n$ កំណត់លើ $\mathbb{R}^*$ ដោយ៖ $$f_n(x) = \frac{e^{nx}}{x^2}$$

តាងដោយ $(\mathscr{C}_n)$ ជាខ្សែកោងរបស់ $f_n$ ក្នុងតម្រុយអរតូណរម៉ាល់ $(O, \vec{i}, \vec{j})$។

១. គណនា $\displaystyle\lim_{x \to -\infty} f_n(x)$, $\displaystyle\lim_{x \to +\infty} f_n(x)$ និង $\displaystyle\lim_{x \to 0} f_n(x)$៖
  • លីមីតត្រង់ $-\infty$ ៖ $$\lim_{x \to -\infty} f_n(x) = \lim_{x \to -\infty} \frac{e^{nx}}{x^2} = \frac{0}{+\infty} = 0$$ ព្រោះ $n \ge 1$, ពេល $x \to -\infty \implies e^{nx} \to 0$ និង $x^2 \to +\infty$។
    ដូចនេះ $$\boxed{\quad \displaystyle\lim_{x \to -\infty} f_n(x) = 0\quad}$$
  • លីមីតត្រង់ $+\infty$ ៖ $$\lim_{x \to +\infty} f_n(x) = \lim_{x \to +\infty} \frac{e^{nx}}{x^2}= +\infty$$ តាមលក្ខណៈលីមីតប្រៀបធៀប (អ៊ិចស្ប៉ូណង់ស្យែលកើនលឿនជាងពហុធា)
    ដូចនេះ $$\boxed{\quad\lim_{x \to +\infty} \frac{e^{nx}}{x^2} = +\infty\quad}$$
  • លីមីតត្រង់ $0$ ៖ $$\lim_{x \to 0} f_n(x) = \lim_{x \to 0} \frac{e^{nx}}{x^2} = \frac{1}{0^+} = +\infty $$ ដោយសារ $\displaystyle\lim_{x \to 0} e^{nx} = e^0 = 1$ និង $\displaystyle\lim_{x \to 0} x^2 = 0^+$ ។ ដូចនេះ $$\boxed{\quad\displaystyle\lim_{x \to 0} f_n(x) = +\infty}$$
២. សិក្សាមែកអនន្តនៃខ្សែកោង $(\mathscr{C}_n)$៖
  • ដោយ $\displaystyle\lim_{x \to 0} f_n(x) = +\infty$ នាំឱ្យបន្ទាត់ $x = 0$ (អ័ក្សអរដោនេ) ជាអាស៊ីមតូតឈរ។
  • ដោយ $\displaystyle\lim_{x \to -\infty} f_n(x) = 0$ នាំឱ្យបន្ទាត់ $y = 0$ (អ័ក្សអាប់ស៊ីស) ជាអាស៊ីមតូតដេកខាង $-\infty$។
  • ខាង $+\infty$ យើងមាន $\displaystyle\lim_{x \to +\infty} f_n(x) = +\infty$ ត្រូវគណនា $\lim_{x \to +\infty} \frac{f_n(x)}{x}$៖ $$\lim_{x \to +\infty} \frac{f_n(x)}{x} = \lim_{x \to +\infty} \frac{e^{nx}}{x^3} = +\infty$$ ដូចនេះ ខាង $+\infty$ ខ្សែកោង $(\mathscr{C}_n)$ មាន មែកប៉ារ៉ាបូលមានទិសដៅតាមអ័ក្សអរដោនេ $(Oy)$។
៣. បង្ហាញថា $f_n$ មានដេរីវេលើ $]0; +\infty[$ និង $]-\infty; 0[$ រួចគណនា $f_n'(x)$៖
  • អនុគមន៍ $f_n$ ជាផលចែកនៃអនុគមន៍ $u(x) = e^{nx}$ និង $v(x) = x^2$ ដែលមានដេរីវេលើ $\mathbb{R}^*$ ដូចនេះ $f_n$ ក៏មានដេរីវេលើ $]-\infty; 0[$ និង $]0; +\infty[$ ដែរ។
  • គណនាដេរីវេ ដោយប្រើរូបមន្ត $\left(\frac{u}{v}\right)'$ ៖ $$f_n'(x) = \frac{(e^{nx})' \cdot x^2 - e^{nx} \cdot (x^2)'}{(x^2)^2}$$ $$f_n'(x) = \frac{n e^{nx} x^2 - 2x e^{nx}}{x^4}$$ ទាញ $x e^{nx}$ ជាកត្តា សម្រួលនឹង $x$ ភាគបែង៖ $$f_n'(x) = \frac{x e^{nx} (nx - 2)}{x^4} = \frac{e^{nx}(nx - 2)}{x^3}$$
៤. សង់តារាងអថេរភាពនៃ $f_n$៖
  • សិក្សាសញ្ញា $f_n'(x)$៖ ដោយ $e^{nx} \gt 0$ ជានិច្ច នោះសញ្ញានៃ $f_n'(x)$ យកសញ្ញាតាមផលចែក $\frac{nx-2}{x^3}$។ ឫសនៃភាគយកគឺ $x = \frac{2}{n}$។
  • បើ $x \in ]-\infty; 0[ \implies nx - 2 \lt 0$ និង $x^3 \lt 0 \implies f_n'(x)\gt 0$ (អនុគមន៍កើន)។
  • បើ $x \in ]0; \frac{2}{n}[ \implies nx - 2 \lt 0$ និង $x^3 \gt 0 \implies f_n'(x) \lt 0$ (អនុគមន៍ចុះ)។
  • បើ $x \in ]\frac{2}{n}; +\infty[ \implies nx - 2 \gt 0$ និង $x^3 \gt 0 \implies f_n'(x) \gt 0$ (អនុគមន៍កើន)។
  • តម្លៃអប្បបរមាត្រង់ $x = \frac{2}{n}$ គឺ៖ $f_n(\frac{2}{n}) = \frac{e^{n(2/n)}}{(2/n)^2} = \frac{e^2}{4/n^2} = \frac{n^2 e^2}{4}$។

៥. បង្ហាញថាសមីការ $f_n(x) = 1$ មានឫសតែមួយគត់ $x_n$ ក្នុងចន្លោះ $]-\infty; 0[$៖
  • លើចន្លោះ $]-\infty; 0[$ អនុគមន៍ $f_n$ ជាប់ និងកើនដាច់ខាត។
  • រូបភាពនៃចន្លោះ $]-\infty; 0[$ តាមអនុគមន៍ $f_n$ គឺ $]0; +\infty[$។
  • ដោយសារចំនួន $1 \in ]0; +\infty[$ នោះតាមទ្រឹស្ដីបទតម្លៃកណ្ដាល សមីការ $f_n(x) = 1$ មានឫសតែមួយគត់ $x_n$ ក្នុងចន្លោះ $]-\infty; 0[$។
៦. បង្ហាញថា $(\forall n \ge 2) \quad -1 \lt x_n \lt -\frac{1}{n}$៖
  • យើងមាន $f_n(x_n) = 1$។ គណនាតម្លៃ $f_n(-1)$ និង $f_n(-\frac{1}{n})$៖
    • $f_n(-1) = \frac{e^{-n}}{(-1)^2} = e^{-n} = \frac{1}{e^n}$ ។ ដោយ $n \ge 2 \implies e^n \gt 1 \implies f_n(-1) \lt 1$ ។
    • $f_n(-\frac{1}{n}) = \frac{e^{n(-1/n)}}{(-1/n)^2} = \frac{e^{-1}}{1/n^2} = \frac{n^2}{e}$ ។ ដោយ $n \ge 2 \implies n^2 \ge 4$ និង $e \approx 2.718 \implies \frac{n^2}{e} \gt 1 \implies f_n(-\frac{1}{n}) \gt 1$ ។
  • យើងបាន $f_n(-1) \lt f_n(x_n) \lt f_n(-\frac{1}{n})$។ ដោយសារ $f_n$ ជាអនុគមន៍កើនដាច់ខាតលើ $]-\infty; 0[$ យើងទាញបាន $-1 \lt x_n \lt -\frac{1}{n}$ ។
៧. បង្ហាញថា $f_{n+1}(x_n) = e^{x_n}$ រួចទាញបញ្ជាក់ថា $(x_n)_n$ ជាស៊ីតរួម៖
  • ដោយ $f_n(x_n) = 1 \implies \frac{e^{nx_n}}{x_n^2} = 1 \implies x_n^2 = e^{nx_n}$ ។
  • គណនា $f_{n+1}(x_n)$ ៖ $$f_{n+1}(x_n) = \frac{e^{(n+1)x_n}}{x_n^2} = \frac{e^{nx_n} \cdot e^{x_n}}{x_n^2} = \frac{x_n^2 \cdot e^{x_n}}{x_n^2} = e^{x_n}$$
  • ទាញបញ្ជាក់ថា $(x_n)$ ជាស៊ីតរួម៖
    ដោយ $x_n \lt 0 \implies e^{x_n} \lt e^0 = 1$។ ដូចនេះ $f_{n+1}(x_n) \lt 1$ ។
    តែយើងដឹងថា $f_{n+1}(x_{n+1}) = 1$ ហេតុនេះ $f_{n+1}(x_n) \lt f_{n+1}(x_{n+1})$ ។
    ដោយសារ $f_{n+1}$ ជាអនុគមន៍កើនដាច់ខាតលើ $]-\infty; 0[$ យើងទាញបាន $x_n \lt x_{n+1}$។
    សន្និដ្ឋាន៖ ស៊ីត $(x_n)$ ជាស៊ីតកើន និងទាល់លើដោយ $0$ (ព្រោះ $x_n \lt 0$) ដូចនេះ $(x_n)_n$ ជាស៊ីតរួម។
៧. ស្រាយបញ្ជាក់ថា $(\forall n \ge 2) \quad x_n \ge -\frac{2\ln n}{n}$ រួចទាញរកលីមីត៖
  • ពីសមីការ $x_n^2 = e^{nx_n}$ បញ្ចូលលោការីតនេពែសងខាង៖ $$\ln(x_n^2) = \ln(e^{nx_n}) \implies 2\ln(|x_n|) = nx_n$$ ដោយ $x_n \lt 0 \implies |x_n| = -x_n \implies 2\ln(-x_n) = nx_n$ ។
  • តាមសំណួរទី៦ យើងមាន $x_n \lt -\frac{1}{n} \implies -x_n \gt \frac{1}{n}$ ។
    បញ្ចូល $\ln$ សងខាង៖ $\ln(-x_n) \gt \ln(\frac{1}{n}) \implies \ln(-x_n) \gt -\ln n$ ។
  • គុណនឹង $2$ ៖ $2\ln(-x_n) \gt -2\ln n \implies nx_n \gt -2\ln n \implies$ $x_n \ge -\frac{2\ln n}{n}$ ។
  • ទាញរកលីមីត $\lim_{n \to +\infty} x_n$ ៖
    យើងមាន $-\frac{2\ln n}{n} \le x_n \lt -\frac{1}{n}$។ តាមទ្រឹស្ដីបទតម្លៃកៀប ដោយ $\lim_{n \to +\infty} \frac{\ln n}{n} = 0$ និង $\lim_{n \to +\infty} \frac{-1}{n} = 0$
    ដូចនេះ $\lim_{n \to +\infty} x_n = 0$ ។
  • ទាញរកលីមីត $\lim_{n \to +\infty} nx_n$ ៖
    យើងមាន $nx_n = 2\ln(-x_n)$។ ដោយ $x_n \to 0$ ពីឆ្វេង នោះ $-x_n \to 0^+$។
    នាំឱ្យ $\lim_{n \to +\infty} \ln(-x_n) = -\infty$ ។
    ដូចនេះ $\lim_{n \to +\infty} nx_n = -\infty$ ។
